7-8 刁总的游戏 题解 : 

题目理解 : 1 .可以将本题抽象看为是图论中的最短路问题 , 将关卡看为节点 , 两种操作对应两条边, 秒数对应边的权			值,边权均为正数 , 使用 Dijkstra算法 .

&nbsp;			2.A, B 范围较大, 使用邻接表来实现算法

&nbsp;		  3.本题注意点 B 走法中Xi可能为本身 , 或者直接跳到N关卡( 要求过全部关卡 ) ; 

AC代码 : 

\#include <iostream>

\#include <cstring>

\#include <queue>

\#include <vector>

using namespace std;



\#define int long long  // 注意数据范围



typedef pair<int, int> PII;

const int N = 2e5 + 10; 



int h\[N], e\[N \* 2], ne\[N \* 2], idx;

int w\[N \* 2];  // 边的权重

int dist\[N];   // 记录到每个节点最短距离

bool visited\[N];  // 防止数据重复进堆 , 优化代码



// 添加有向边

void add(int a, int b, int c) {

&nbsp;   e\[idx] = b;

&nbsp;   ne\[idx] = h\[a];

&nbsp;   w\[idx] = c;

&nbsp;   h\[a] = idx++;

}



// Dijkstra算法求最短路

int dijkstra(int n) {

&nbsp;   memset(dist, 0x3f, sizeof dist);

&nbsp;   dist\[1] = 0;

&nbsp;   

&nbsp;   // 使用小根堆，按距离排序

&nbsp;   priority\_queue<PII, vector<PII>, greater<PII>> heap;

&nbsp;   heap.push({0, 1});  // {距离, 节点}

&nbsp;   

&nbsp;   while (!heap.empty()) {

&nbsp;       auto t = heap.top();

&nbsp;       heap.pop();

&nbsp;       

&nbsp;       int d = t.first, u = t.second;

&nbsp;       

&nbsp;       if (visited\[u]) continue;

&nbsp;       visited\[u] = true;

&nbsp;       

&nbsp;       // 遍历所有邻接边

&nbsp;       for (int i = h\[u]; i != -1; i = ne\[i]) {

&nbsp;           int v = e\[i];

&nbsp;           if (dist\[v] > d + w\[i]) {          //找出最短距离 ; 

&nbsp;               dist\[v] = d + w\[i];

&nbsp;               heap.push({dist\[v], v});

&nbsp;           }

&nbsp;       }

&nbsp;   }

&nbsp;   

&nbsp;   return dist\[n];

}



signed main() { 

&nbsp;   ios::sync\_with\_stdio(false);

&nbsp;   cin.tie(nullptr);

&nbsp;   

&nbsp;   int N;

&nbsp;   cin >> N;

&nbsp;   

&nbsp;   // 初始化邻接表

&nbsp;   memset(h, -1, sizeof h);

&nbsp;   idx = 0;

&nbsp;   for (int i = 1; i < N; i++) {

&nbsp;       int A, B, X; 

&nbsp;       cin >> A >> B >> X;

&nbsp;       add(i, i + 1, A);  // 第一种操作：i -> i+1，时间A

&nbsp;       add(i, X, B);      // 第二种操作：i -> X，时间B

&nbsp;   }

&nbsp;   int result = dijkstra(N);

&nbsp;   cout << result << endl;

&nbsp;   

&nbsp;   return 0;

}
