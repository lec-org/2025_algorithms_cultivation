const int N = 1e3 + 10;
int a[N], b[N];  // a：维护递减序列；b：维护递增序列
int n, x, t1, t2, num, mx;  // t1/t2：a/b数组的（下一个插入位置）；num：段数；mx：最长段长度
int main() {
    cin >> n;
    t1 = 0, t2 = 0;  // 指针初始化为0（数组为空）
    num = 0; mx = 0; // 段数、最长长度初始化为0
    cin >> a[t1++];  // 第一个数直接放入a数组，t1变为1
for (int i = 1; i < n; ++i) {  // 处理第2到第n个数
    cin >> x;
    // 情况1：x比a最后一个数小 → 接在a后面（保持a递减）
    if (x < a[t1 - 1]) {
        a[t1++] = x;
    } else {
        // 情况2：x能接在b后面（b为空 或 x比b最后一个数大）→ 接在b后面（保持b递增）
        if (t2 == 0 || x > b[t2 - 1]) {
            b[t2++] = x;
        } else {
            // 情况3：x既不能接a，也不能接b → 拆分新段
            num++;  // 旧段计数+1
            mx = max(mx, t1);  // 更新最长段长度（旧段的有效长度是t1）
            t1 = 0;  // 重置a数组，准备新段
            // 把b中比x大的数移到新的a数组里
            while (t2 > 0 && b[t2 - 1] > x) {
                a[t1++] = b[--t2];
            }
            a[t1++] = x;  // x放入新的a数组，作为新段的起点
        }
    }
}
// 遍历结束后，处理剩下的未计数的段
if (t1 > 0) {  // a数组有元素 → 这是最后一段，计数+1，更新最长长度
    num++;
    mx = max(mx, t1);
}
if (t2 > 0) {  // b数组有元素 → 单独算一段
    num++;
    mx = max(mx, t2);
}
cout << num << " " << mx << endl;

return 0；
}
