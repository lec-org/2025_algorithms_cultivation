**题目：给定一个整数 n，请你计算 n 各位数字的和，直到数字和是一个一位数为止。也就是说，你需要不断地将数字各位之和计算出来，直到结果是一个一位数。**

\#include <bits/stdc++.h>

using namespace std;

int t;

string n;                                  //由于输入的数据为10的100次方，数据过大，因此采用字符串输入方式

int find (int sum) {

    int a=0;

    while (sum/10>0) {

        a+=sum%10;

        sum/=10;

    }

    a+=sum;

    return a;

}                                             //构造一个函数find去进行不断地将数字各位之和计算

int main()

{

    cin >> t;

    while (t--) {

        int sum=0;

        cin >> n;

        for (int i=0;i<n.length();i++) {

            sum+=n\[i]-'0';

        }                                    //将输入的n转换为一个最大为900的第一次将数字各位之和的计算操作

        while (sum>9) {

            sum=find(sum);      //不断地将数字各位之和计算

        }

        cout << sum << endl;

    }

 

    return 0;

}
