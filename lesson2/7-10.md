搜索+剪枝
从左到右依次搜索，到n+1则提行
剪掉过大或者过小的枝，于是需要srow,scol来存储每一行，每一列的前缀和啊，如果前缀和过大或者过小则减掉这条枝
没减掉的枝一定是吉利矩阵，不用重新验证。
```cpp
#include <bits/stdc++.h>
const int N = 11;
using namespace std;
using ll = long long;
int scol[N],srow[N];
int l,n;
ll cnt = 0;
void dfs(int x,int y){
    if(y == n+1){
        x++;
        y = 1;
    }
    if(x == n+1){
        cnt++;
        return ;
    }
    for(int i = 0;i <= l;i++){
        if(i + scol[x] > l || i + srow[y] > l) continue;
        if(i + scol[x] + (n-y)*l < l || i + srow[y] + (n-x)*l < l) continue;
        scol[x] += i;
        srow[y] += i;
        dfs(x,y+1);
        scol[x] -= i;
        srow[y] -= i;
    }
}
int main(){
    cin >> l >> n;
    dfs(1,1);
    cout << cnt;
}
```
