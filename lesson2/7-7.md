# 解题思路
使用一个二维数组来存储地图，遍历这个地图，当遍历的节点是陆地且未被访问过就启动一次**DFS**。通过DFS遍历掉整个地图。
# 注意事项
由于题目的数据要求是：1<N×M≤$10^5$，直接开辟普通二维数组会爆内存，所以使用**vector**来进行存储。
# 代码实现
```c++
#include <iostream>
#include <vector>
#include <string>

using namespace std;

const int n = 1e5 + 5, m = 1e5 + 5;
vector < vector <int> > map;
// 二维数组存储地图
vector < vector <bool> > visit;
// 二维数组用于判断地图的各个节点是否遍历过
// 注意：最好多打几个空格，防止代码被识别成">>"运算符
int mid[4][2] = { { -1, 0 }, { 0, -1 }, { 1, 0 }, { 0, 1 } };
// 每个节点的上下左右
int N, M;
int cnt, ret;// cnt是岛屿数量，ret是有宝藏的岛屿的数量
bool flag;// 判断是否有宝藏

void dfs ( int x, int y ) {
    if ( x < 0 || x >= N || y < 0 || y >= M || visit[x][y] || !map[x][y] ) {
        return;
    }
    // 如果数据越界或者被遍历过或者遇到海洋，本次DFS结束
    // 注意：如果把判断条件写进同一个if语句内，要先判断是否越界，会出现段错误
    visit[x][y] = true;
    // 标记此节点被遍历过
    if ( map[x][y] >= 2 && map[x][y] <= 9 ) {
        flag = true;
    }
    // 找到宝藏
    for ( int i = 0; i < 4; ++ i ) {
        dfs ( x + mid[i][0], y + mid[i][1] );
    }
    // 对该节点的上下左右四个格子进行DFS
}

int main ( ) {
    cin >> N >> M;
    map.resize ( N, vector <int> (M) );
    visit.resize ( N, vector <bool> (M, false) );
    // resize是vector内函数，用于动态调整容器大小
    for ( int i = 0; i < N; ++ i ) {
        string s;
        cin >> s;
        for ( int j = 0; j < M; ++ j ) {
            map[i][j] = s[j] - '0';
        }
    }
    //存储地图，由于给的数据是连续字符串，所以不适用两个for循环读入的代码。
    for ( int i = 0; i < N; ++ i ) {
        for ( int j = 0; j < M; ++ j ) {
            if ( map[i][j] && !visit[i][j] ) {
                ++ cnt;
                flag = false;// 初始化
                dfs ( i, j );// 遍历此节点所在的整个岛屿
                if ( flag ) ++ ret;
            }
            // 若当前节点是陆地且未被访问过，启动DFS
        }
    }
    cout << cnt << " " << ret << endl;
}
```
